cmake_minimum_required(VERSION 3.15)
project(Dedcon VERSION 1.0)

option(BUILD_RCON_CLIENT "Build the RCON client" ON)
option(BUILD_DEDCON_CLIENT "Build the Dedcon client" ON)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(OPENSSL_ROOT_DIR "${CMAKE_SOURCE_DIR}/OpenSSL")
set(OPENSSL_INCLUDE_DIR "${OPENSSL_ROOT_DIR}/include")
if(WIN32)
    set(OPENSSL_LIBRARIES_DIR "${OPENSSL_ROOT_DIR}/lib/VC/Windows")
elseif(APPLE)
    set(OPENSSL_LIBRARIES_DIR "${OPENSSL_ROOT_DIR}/lib/VC/MacOS")
else()
    set(OPENSSL_LIBRARIES_DIR "${OPENSSL_ROOT_DIR}/lib/VC/Linux")
endif()
include_directories(${OPENSSL_INCLUDE_DIR})

if(WIN32)
    add_definitions(-DOPENSSL_STATIC -DWIN32_LEAN_AND_MEAN)
else()
    add_definitions(-DOPENSSL_STATIC)
endif()

if(APPLE)
    set(CMAKE_OSX_ARCHITECTURES "x86_64;arm64")
    set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_SOURCE_DIR}/bin")
    set(CMAKE_CONFIGURATION_TYPES "Release Vanilla;Release Eight Player;Release Ten Player;Debug Vanilla;Debug Eight Player;Debug Ten Player")
else()
    set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_SOURCE_DIR}/DedconExecutables")
    set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
    set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
    set(CMAKE_CONFIGURATION_TYPES "Release Vanilla;Release Eight Player;Release Ten Player;Debug Vanilla;Debug Eight Player;Debug Ten Player" CACHE STRING "" FORCE)
    if(WIN32)
        set(CMAKE_GENERATOR_PLATFORM Win32)
    endif()
endif()

if(MSVC)
    add_definitions(-D_WIN32 -DWIN32 -D_WINSOCK_DEPRECATED_NO_WARNINGS -D_CRT_SECURE_NO_WARNINGS)
    set(CMAKE_GENERATOR_TOOLSET "v142")
    set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} /SUBSYSTEM:CONSOLE /STACK:8388608")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /MP /EHsc /Zm2000 /Zf")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /wd4530 /wd4018 /wd4267 /wd4244 /wd4305 /wd4101 /wd4099 /wd26495 /wd4146")
elseif(APPLE)
    add_definitions(-D_DARWIN -DMACOS)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -Wno-deprecated-declarations")
elseif(UNIX AND NOT APPLE)
    add_definitions(-D_LINUX -DLINUX)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -Wno-deprecated-declarations")
endif()

foreach(CONFIG ${CMAKE_CONFIGURATION_TYPES})
    string(TOUPPER ${CONFIG} CONFIG_UPPER)
    
    if(MSVC)
        set(CMAKE_CXX_FLAGS_${CONFIG_UPPER} "${CMAKE_CXX_FLAGS_${CONFIG_UPPER}} /MP")
    endif()
    
    if(${CONFIG} MATCHES "Debug")
        if(MSVC)
            string(APPEND CMAKE_CXX_FLAGS_${CONFIG_UPPER} " /Od /Zi /MDd /RTC1 /GS /EHsc /FC /analyze /Z7")
            set(CMAKE_C_FLAGS_${CONFIG_UPPER} "${CMAKE_C_FLAGS_${CONFIG_UPPER}} /MDd -D_DEBUG -DDEBUG -DDEBUG_X -DNDEBUG -DXDEBUG")
            set(CMAKE_CXX_FLAGS_${CONFIG_UPPER} "${CMAKE_CXX_FLAGS_${CONFIG_UPPER}} /MDd -D_DEBUG -DDEBUG -DDEBUG_X -DNDEBUG -DXDEBUG")
        else()
            set(CMAKE_CXX_FLAGS_${CONFIG_UPPER} "${CMAKE_CXX_FLAGS_${CONFIG_UPPER}} -O0 -g")
            set(CMAKE_C_FLAGS_${CONFIG_UPPER} "${CMAKE_C_FLAGS_${CONFIG_UPPER}} -D_DEBUG -DDEBUG -DDEBUG_X -DNDEBUG -DXDEBUG")
            set(CMAKE_CXX_FLAGS_${CONFIG_UPPER} "${CMAKE_CXX_FLAGS_${CONFIG_UPPER}} -D_DEBUG -DDEBUG -DDEBUG_X -DNDEBUG -DXDEBUG")
        endif()
    else()
        if(MSVC)
            string(APPEND CMAKE_CXX_FLAGS_${CONFIG_UPPER} " /MD /O2")
            set(CMAKE_C_FLAGS_${CONFIG_UPPER} "${CMAKE_C_FLAGS_${CONFIG_UPPER}} /MD")
            set(CMAKE_CXX_FLAGS_${CONFIG_UPPER} "${CMAKE_CXX_FLAGS_${CONFIG_UPPER}} /MD")
        else()
            set(CMAKE_CXX_FLAGS_${CONFIG_UPPER} "${CMAKE_CXX_FLAGS_${CONFIG_UPPER}} -O3")
        endif()
    endif()

    if(${CONFIG} MATCHES "Eight Player")
        set(CMAKE_C_FLAGS_${CONFIG_UPPER} "${CMAKE_C_FLAGS_${CONFIG_UPPER}} -DEIGHTPLAYER")
        set(CMAKE_CXX_FLAGS_${CONFIG_UPPER} "${CMAKE_CXX_FLAGS_${CONFIG_UPPER}} -DEIGHTPLAYER")
    elseif(${CONFIG} MATCHES "Ten Player")
        set(CMAKE_C_FLAGS_${CONFIG_UPPER} "${CMAKE_C_FLAGS_${CONFIG_UPPER}} -DTENPLAYER")
        set(CMAKE_CXX_FLAGS_${CONFIG_UPPER} "${CMAKE_CXX_FLAGS_${CONFIG_UPPER}} -DTENPLAYER")
    elseif(${CONFIG} MATCHES "Vanilla")
        set(CMAKE_C_FLAGS_${CONFIG_UPPER} "${CMAKE_C_FLAGS_${CONFIG_UPPER}} -DVANILLA")
        set(CMAKE_CXX_FLAGS_${CONFIG_UPPER} "${CMAKE_CXX_FLAGS_${CONFIG_UPPER}} -DVANILLA")
    endif()

    set(CMAKE_C_FLAGS_${CONFIG_UPPER} "${CMAKE_C_FLAGS_${CONFIG_UPPER}} -DCOMPILE_DEDCON")
    set(CMAKE_CXX_FLAGS_${CONFIG_UPPER} "${CMAKE_CXX_FLAGS_${CONFIG_UPPER}} -DCOMPILE_DEDCON")
endforeach()

if(BUILD_DEDCON_CLIENT)
    add_executable(Dedcon ${DEDCON_SOURCES} ${WIN32_RESOURCES})

    if(WIN32)
        target_link_libraries(Dedcon ws2_32)
        if(MSVC)
            target_link_libraries(Dedcon debug
                ${OPENSSL_LIBRARIES_DIR}/MTd/libssl_static.lib
                ${OPENSSL_LIBRARIES_DIR}/MTd/libcrypto_static.lib
                crypt32.lib
                advapi32.lib
                user32.lib
            )
            target_link_libraries(Dedcon optimized
                ${OPENSSL_LIBRARIES_DIR}/MT/libssl_static.lib
                ${OPENSSL_LIBRARIES_DIR}/MT/libcrypto_static.lib
                crypt32.lib
                advapi32.lib
                user32.lib
            )
        endif()
    elseif(APPLE)
        target_link_libraries(Dedcon 
            ${OPENSSL_LIBRARIES_DIR}/MT/libssl.a
            ${OPENSSL_LIBRARIES_DIR}/MT/libcrypto.a
            ${OPENSSL_LIBRARIES_DIR}/MT/libssl_arm64.a
            ${OPENSSL_LIBRARIES_DIR}/MT/libcrypto_arm64.a
            dl pthread
        )
    elseif(UNIX)
        target_link_libraries(Dedcon 
            ${OPENSSL_LIBRARIES_DIR}/MT/libssl_static.lib
            ${OPENSSL_LIBRARIES_DIR}/MT/libcrypto_static.lib
            dl pthread
        )
    endif()

    if(MSVC)
        target_compile_options(Dedcon PRIVATE /utf-8)
        set_property(TARGET Dedcon PROPERTY MSVC_RUNTIME_LIBRARY "MultiThreadedDebugDLL")
    endif()

    if(APPLE)
        set_target_properties(Dedcon PROPERTIES 
            OUTPUT_NAME_RELEASE_VANILLA "Dedcon"
            OUTPUT_NAME_RELEASE_EIGHT_PLAYER "Dedcon8p"
            OUTPUT_NAME_RELEASE_TEN_PLAYER "Dedcon10p"
            OUTPUT_NAME_DEBUG_VANILLA "Dedcon"
            OUTPUT_NAME_DEBUG_EIGHT_PLAYER "Dedcon8p"
            OUTPUT_NAME_DEBUG_TEN_PLAYER "Dedcon10p"
        )
    else()
        set_target_properties(Dedcon PROPERTIES "RELEASE_VANILLA_OUTPUT_NAME" "Dedcon")
        set_target_properties(Dedcon PROPERTIES "RELEASE_EIGHT_PLAYER_OUTPUT_NAME" "Dedcon8p")
        set_target_properties(Dedcon PROPERTIES "RELEASE_TEN_PLAYER_OUTPUT_NAME" "Dedcon10p")
        set_target_properties(Dedcon PROPERTIES "DEBUG_VANILLA_OUTPUT_NAME" "Dedcon")
        set_target_properties(Dedcon PROPERTIES "DEBUG_EIGHT_PLAYER_OUTPUT_NAME" "Dedcon8p")
        set_target_properties(Dedcon PROPERTIES "DEBUG_TEN_PLAYER_OUTPUT_NAME" "Dedcon10p")
    endif()

    set_target_properties(Dedcon PROPERTIES "RELEASE_VANILLA_OUTPUT_NAME" "Dedcon")
    set_target_properties(Dedcon PROPERTIES "RELEASE_EIGHT_PLAYER_OUTPUT_NAME" "Dedcon8p")
    set_target_properties(Dedcon PROPERTIES "RELEASE_TEN_PLAYER_OUTPUT_NAME" "Dedcon10p")
    set_target_properties(Dedcon PROPERTIES "DEBUG_VANILLA_OUTPUT_NAME" "Dedcon")
    set_target_properties(Dedcon PROPERTIES "DEBUG_EIGHT_PLAYER_OUTPUT_NAME" "Dedcon8p")
    set_target_properties(Dedcon PROPERTIES "DEBUG_TEN_PLAYER_OUTPUT_NAME" "Dedcon10p")

    install(TARGETS Dedcon DESTINATION bin)
endif()
