# Set the server name
ServerName Tournament Game

# Set a password
ServerPassword abc

# Activate tournament mode to get all the default settings
#GameMode 5

# Sleep for a second to activate the tournament settings
WaitSeconds 1

# Switch over to default mode so we can make the spectator chat public
GameMode 0

# Make the spectator channel public
SpectatorChatChannel 1

# Set the number of players here
MaxTeams 3

# Wait for the game to start
WaitStart

# Switch back to tournament mode. This makes the spectator chat channel private automatically.
GameMode 5

# Disable the password so spectators can join.
ServerPassword

# Wait for the game to end
WaitEnd

# Wait up to one hour
WaitSeconds 3600

# And quit, irregardless of spectators
Quit
